#include<stdio.h>


void main()
{


int i,j,m,n,p,r,s=0,reqp,request[10],safe[10],alloc[10][10],need[10][10],max[10][10],finish[10],work[10],avail[10];;
   printf("\n Enter the how many process you want:");
   scanf("%d",&p);
   printf(" Enter the how many resources you want:");
   scanf("%d",&r);
   printf("\n Enter the allocation matrix:\n");
   for(i=0;i<p;i++)
    {
     finish[i]=-1;
     printf(" P%d\t",i);
      for(j=0;j<r;j++)
       scanf("%d",&alloc[i][j]);
    }
   printf("\n Enter the maximum demand of matrix:\n");
   for(i=0;i<p;i++)
    {
     printf(" P%d\t",i);
      for(j=0;j<r;j++)
       scanf("%d",&max[i][j]);
    }
   printf("\n Enter the available resource matrix: ");
   for(i=0;i<r;i++)
     {
       scanf("%d",&avail[i]);
       work[i]=avail[i];
     }

  find_avail_need();
   printf("\nEnter process id for a request:");
 scanf("%d",&reqp);
   printf("\nEnter request for %d resources:",m);
   for (r=0; r<m; r++)
 scanf("%d",&request[r]);
   s=0;
   resource_request();
}

void find_avail_need()
{
	
	   printf("\n\n Content of need array is...\n");
   for (p=0; p<n; p++)
   {
	for (r=0; r<m; r++)
	{
	  need[p][r]=max[p][r]-alloc[p][r];
	   printf("\t%d\t",need[p][r]);
	  }
	  printf("\n");
}


 }

void resource_request()
{
  
   for (r=0; r<m; r++)
	 if (request[r]>need[reqp][r])
		{ printf("Process has exceeded its maximum claim");
		 
		  exit(0);
		}
	
   for (r=0; r<m; r++)
	 if (request[r]>avail[r])
		{ printf("Resources are not available Process should wait..");
		  
		  exit(0);
		}
	
	  for (r=0; r<m; r++)
		{ avail[r]-=request[r];
		  alloc[reqp][r]+=request[r];
		  need[reqp][r]-=request[r];
		}
	
	 if (safety())
	   {  printf("\nRequest can be immediately granted...");
		  print_safeseq();
	   }
	 else
	   printf("\nRequest can not be immediately granted...");
	 

}


int safety()
 {
   
   for (r=0; r<m; r++)
	  work[r]=avail[r];
   for (p=0; p<n; p++)
	  finish[p]=0;
  while(1)
  {
 
	p=find_process();
	if (p==-1) 
	  {
		for (p=0; p<n; p++)
		   if(finish[p]==0)
			   return(0);
		  return(1);
	   } 
	else 
	{   
		 safeseq[s++]=p;
		
		for (r=0; r<m; r++)
		   work[r]+=alloc[p][r];
		
		  finish[p]=1;
	}
  }
 }

void print_safeseq()
{
   printf("\nSafe Seq:");
   for(p=0; p<s; p++)
	 printf("%d\t",safeseq[p]);
}

int find_process()
 {  int found;
	for (p=0; p<n; p++)
	{  if (finish[p]==0)
		{ found=1;
		  for (r=0; r<m; r++)
			 if (need[p][r]>work[r])
			   { found=0; break;}
		   if (found)
			return(p);
		 }
	  }
	  return(-1);
 }
