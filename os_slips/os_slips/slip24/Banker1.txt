//slip no. 3,13,19,24,26

#include<stdio.h>
#include<math.h>

int n,m,p,r;
int avail[4]={1,5,2,0};
int max[][4]={{0,0,1,2},{1,7,5,0},{2,3,5,6},{0,6,5,2},{0,6,5,6}};
int alloc[][4]={{0,0,1,2},{1,0,0,0},{1,3,5,4},(0,6,3,2},{0,0,1,4}};
int need[10][10];




void main()
{

 
  n=5; m=4;
 find_avail_need();
  
  if (safety())
	{
	  printf("Current allocation state is safe");
	  print_safeseq();
	 }
   else
	  printf("Current allocation state is not safe");

}

void find_avail_need()
{
	
   for (p=0; p<n; p++)
   {
	for (r=0; r<m; r++)
	{
	  need[p][r]=max[p][r]-alloc[p][r];
	   printf("\t%d\t",need[p][r]);
	  }
	  printf("\n");
}


 }

int safety()
 {
   
   for (r=0; r<m; r++)
	  work[r]=avail[r];
   for (p=0; p<n; p++)
	  finish[p]=0;
  while(1)
  {
 
	p=find_process();
	if (p==-1) 
	  {
		for (p=0; p<n; p++)
		   if(finish[p]==0)
			   return(0);
		  return(1);
	   } 
	else 
	{   
		 safeseq[s++]=p;
		
		for (r=0; r<m; r++)
		   work[r]+=alloc[p][r];
		
		  finish[p]=1;
	}
  }
 }

void print_safeseq()
{
   printf("\nSafe Seq:");
   for(p=0; p<s; p++)
	 printf("%d\t",safeseq[p]);
}

int find_process()
 {  int found;
	for (p=0; p<n; p++)
	{  if (finish[p]==0)
		{ found=1;
		  for (r=0; r<m; r++)
			 if (need[p][r]>work[r])
			   { found=0; break;}
		   if (found)
			return(p);
		 }
	  }
	  return(-1);
 }
