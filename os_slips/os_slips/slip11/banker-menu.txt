//slip no. 
#include<stdio.h>
#include<math.h>

int n,m,p,r;
int avail[4];
int max[][4]={{0,0,1,2},{1,7,5,0},{2,3,5,6},{0,6,5,2},{0,6,5,6}};
int alloc[][4]={{0,0,1,2},{1,0,0,0},{1,3,5,4},(0,6,3,2},{0,0,1,4}};
int need[10][10];



void main()
{

int ch;
do
{
printf("\n 1.  Accept Available");
printf("\n 2.  Display Allocation, Max");
printf("\n 3.   Display the contents of need matrix");
printf("\n  4. Display Available");
printf("\n  5. Exit");

printf("\n Enter choice:");
scanf("%d",&ch);
switch (ch)
{
case 1:    printf("\n Enter the available resource matrix: ");
   for(i=0;i<r;i++)
     {
       scanf("%d",&avail[i]);
}
break;
case 2: printf("\n  allocation matrix:\n");
   for(i=0;i<p;i++)
    {
     
     printf(" P%d\t",i);
      for(j=0;j<r;j++)
      printf("\t%d",alloc[i][j]);
printf("\n");

    }
   printf("\n Enter the maximum demand of matrix:\n");
   for(i=0;i<p;i++)
    {
     printf(" P%d\t",i);
      for(j=0;j<r;j++)
       printf("\t%d",max[i][j]);
printf("\n");
    }
break;
case 3:	   printf("\n\n Content of need array is...\n");
   for (p=0; p<n; p++)
   {
	for (r=0; r<m; r++)
	{
	  need[p][r]=max[p][r]-alloc[p][r];
	   printf("\t%d\t",need[p][r]);
	  }
	  printf("\n");
}
break;
case 4:printf("\n available resource matrix: ");
   for(i=0;i<r;i++)
     
       printf("%d",avail[i]);
break;
case 5: exit(0);
}
}while(ch!=5);

}
