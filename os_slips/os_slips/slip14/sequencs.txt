#include <stdlib.h>
#include <string.h>

#define SIZE 100
#define NEWNODE (struct direntry*) malloc(sizeof(struct direntry))

struct direntry
{
    char fname[14];
    int start;
    int count;
    struct direntry *next;
} *dirlst = NULL, *dirend = NULL;

int bitvector[SIZE];

void main()
{
    int ch = 0, i, j, k, n, flag, flag1;
    char fname[14], f[15];
    struct direntry *t1, *t2;

    for (i = 0; i < SIZE; i++)
    {
        bitvector[i] = rand() % 2;
    }

    while (1)
    {
        printf("\n\n1. print bit vector");
        printf("\n2. create file");
        printf("\n3. print directory");
        printf("\n4. delete file");
    

	printf("\n5. Exit");
	printf("\nEnter your choice");
	scanf("%d", &ch);

switch (ch)
{
    case 1:
        for (i = 0; i < SIZE; i++)
            printf("%d", bitvector[i]);
        break;

    case 2:
        if (dirlst == NULL)
            dirlst = dirend = NEWNODE;
        else
        {
            dirend->next = NEWNODE;
            dirend = dirend->next;
        }

        dirend->next = NULL;

        printf("\nEnter a filename:");
        scanf("%s", f);

        printf("\nEnter the number of blocks to allocate");
        scanf("%d", &n);

        flag = 0;

        for (i = 0; i < SIZE; i++)
        {
            if (bitvector[i] == 1)
            {
                for (j = i; j < i + n; j++)
                {
                    if (bitvector[j] != 1)
                        break;
                }
           if(j==i+n)

	{
    flag1 = 1;
    dirend->start = i;

    for (k = i; k < j; k++)
        bitvector[k] = 0;

    break;
}

if (flag1 == 1)
{
    strcpy(dirend->fname, f);
    dirend->count = n;
}
else
{
    printf("\nSufficient blocks not available");
}
break;

case 3:
    printf("\nDirectory:");
    printf("\n----------------");
    printf("\nfilename   start   count");
    printf("\n----------------");

    for (t1 = dirlst; t1 != NULL; t1 = t1->next)
        printf("\n%-10s %5d %5d", t1->fname, t1->start, t1->count);

    printf("\n----------------");
    break;

case 4:
    printf("\nEnter filename:");
    scanf("%s", fname);
	t1 = dirlst;

while (t1 != NULL)
{
    if (strcmp(t1->fname, fname) == 0)
        break;

    t2 = t1;
    t1 = t1->next;
}

if (t1 != NULL)
{
    for (i = t1->start; i < t1->start + t1->count; i++)
        bitvector[i] = 1;

    if (t1 == dirlst)
        dirlst = dirlst->next;
    else
        t2->next = t1->next;

    if (dirlst == NULL)
        dirend = NULL;

    free(t1);
}
else
{
    printf("\nFile not found\n");
}
break;

case 5:
    exit(0);
}
}
}
